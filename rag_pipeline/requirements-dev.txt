# ============================================================================
# Usage Instructions
# ============================================================================
#
# Install main dependencies only:
#   pip install -r requirements.txt
#
# Install development dependencies (for local development):
#   pip install -r requirements-dev.txt
#
# Docker users:
#   Development dependencies are NOT needed in Docker
#   The container has everything it needs from requirements.txt
#
# Recommended setup for local development:
#   python3 -m venv venv
#   source venv/bin/activate  # On Windows: venv\Scripts\activate
#   pip install -r requirements-dev.txt
#   pre-commit install  # Optional: Enable git hooks
#
# ============================================================================
# Tool Configuration
# ============================================================================
#
# Black (code formatter):
#   black src/ main.py scripts/
#
# Flake8 (linter):
#   flake8 src/ main.py scripts/
#
# mypy (type checker):
#   mypy src/ main.py --strict
#
# isort (import sorter):
#   isort src/ main.py scripts/
#
# pytest (run tests):
#   pytest tests/ -v --cov=src
#
# ============================================================================
# Pre-commit hooks (optional)
# ============================================================================
#
# Create .pre-commit-config.yaml:
#
# repos:
#   - repo: https://github.com/psf/black
#     rev: 23.7.0
#     hooks:
#       - id: black
#   - repo: https://github.com/pycqa/flake8
#     rev: 6.1.0
#     hooks:
#       - id: flake8
#   - repo: https://github.com/pycqa/isort
#     rev: 5.12.0
#     hooks:
#       - id: isort
#   - repo: https://github.com/pre-commit/mirrors-mypy
#     rev: v1.5.0
#     hooks:
#       - id: mypy
#         additional_dependencies: [types-all]
#
# Then run: pre-commit install
#
# ============================================================================
